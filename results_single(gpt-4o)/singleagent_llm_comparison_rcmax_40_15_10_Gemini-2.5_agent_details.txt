    *   Iterate through the operations, considering them in order of MWR.
    *   For each operation:
        *   Find the earliest time the required machine is available (considering previously scheduled operations).
        *   Ensure the operation's start time is not earlier than the end time of the previous operation in the same job.
        *   Schedule the operation on the machine, updating its start and end times and the machine's availability.

3.  **Local Search (Pairwise Swap):**
    *   Iterate through each machine.
    *   For each pair of operations on the machine, tentatively swap their order.
    *   Calculate the makespan of the resulting schedule.
    *   If the new makespan is lower than the current makespan, accept the swap.
    *   Repeat the pairwise swap process for a fixed number of iterations or until no further improvement is found.

4.  **Makespan Calculation:**
    *   The makespan is the maximum end time of all operations.

5.  **Output:**
    *   Format the schedule and makespan as requested.

```python
import copy

def solve_jssp(job_specifications):
    """Solves the Job Shop Scheduling Problem using a heuristic approach."""

    num_jobs = len(job_specifications)
    num_machines = 15  # Assuming 15 machines based on the problem description

    # 1. Initialization
    operations = []
    for job_name, job_data in job_specifications.items():
        for i, step_data in enumerate(job_data):
            operations.append({
                'Job': job_name,
                'Step': i + 1,
                'Machine': step_data['Machine'],
                'Duration': step_data['Duration'],
                'Start Time': 0,
                'End Time': 0
            })

    machine_availability = {f'Machine{i}': 0 for i in range(num_machines)}
    job_completion_times = {job_name: 0 for job_name in job_specifications}

    # 2. Priority Rule Scheduling (MWR)
    def calculate_mwr(operations, job_completion_times):
        mwr = {}
        for job_name in job_specifications:
            remaining_work = 0
            for op in operations:
                if op['Job'] == job_name and op['End Time'] == 0:
                    remaining_work += op['Duration']
            mwr[job_name] = remaining_work
        return mwr

    scheduled_operations = []
    available_operations = copy.deepcopy(operations)

    while available_operations:
        mwr = calculate_mwr(available_operations, job_completion_times)
        # Find the job with the most work remaining among available operations
        eligible_jobs = set(op['Job'] for op in available_operations)
        if not eligible_jobs:
            break  # No more eligible jobs

        most_work_job = max(eligible_jobs, key=lambda job: mwr[job])

        # Find the first operation of that job that is available
        next_op = next((op for op in available_operations if op['Job'] == most_work_job and op['Step'] == min(op2['Step'] for op2 in available_operations if op2['Job'] == most_work_job)), None)

        if not next_op:
            break

        machine = next_op['Machine']
        duration = next_op['Duration']

        start_time = max(machine_availability[machine], job_completion_times[next_op['Job']])

        next_op['Start Time'] = start_time
        next_op['End Time'] = start_time + duration

        machine_availability[machine] = next_op['End Time']
        job_completion_times[next_op['Job']] = next_op['End Time']

        scheduled_operations.append(next_op)
        available_operations.remove(next_op)

    # 3. Local Search (Pairwise Swap)
    def calculate_makespan(scheduled_operations):
        return max(op['End Time'] for op in scheduled_operations)

    def perform_swap(ops, i, j):
        ops[i], ops[j] = ops[j], ops[i]

    def revert_swap(ops, i, j):
        ops[i], ops[j] = ops[j], ops[i]

    def reschedule_machine(machine_name, scheduled_operations):
        machine_ops = sorted([op for op in scheduled_operations if op['Machine'] == machine_name], key=lambda x: x['Start Time'])
        availability = 0
        for op in machine_ops:
            op['Start Time'] = max(availability, job_completion_times[op['Job']])
            op['End Time'] = op['Start Time'] + op['Duration']
            availability = op['End Time']
            job_completion_times[op['Job']] = op['End Time']

    num_iterations = 100
    original_makespan = calculate_makespan(scheduled_operations)
    best_makespan = original_makespan
    best_schedule = copy.deepcopy(scheduled_operations)

    for _ in range(num_iterations):
        for machine_name in machine_availability:
            machine_ops = [op for op in scheduled_operations if op['Machine'] == machine_name]
            num_ops = len(machine_ops)

            for i in range(num_ops):
                for j in range(i + 1, num_ops):
                    # Tentatively swap operations
                    perform_swap(machine_ops, i, j)

                    # Recalculate start and end times for the swapped operations and subsequent operations on the same machine
                    job_completion_times = {job_name: 0 for job_name in job_specifications}
                    reschedule_machine(machine_name, scheduled_operations)

                    # Calculate the new makespan
                    new_makespan = calculate_makespan(scheduled_operations)

                    # If the new makespan is better, keep the swap; otherwise, revert
                    if new_makespan < best_makespan:
                        best_makespan = new_makespan
                        best_schedule = copy.deepcopy(scheduled_operations)
                    else:
                        revert_swap(machine_ops, i, j)
                        job_completion_times = {job_name: 0 for job_name in job_specifications}
                        reschedule_machine(machine_name, scheduled_operations)

    # 4. Makespan Calculation (using the best schedule)
    makespan = best_makespan

    # 5. Output
    structured_schedule = []
    for op in best_schedule:
        structured_schedule.append({
            'Job': op['Job'],
            'Step': op['Step'],
            'Machine': op['Machine'],
            'Start Time': op['Start Time'],
            'End Time': op['End Time'],
            'Duration': op['Duration']
        })

    return makespan, structured_schedule
