    Args:

    Returns:

    for job_name, job_data in job_specifications.items():

            duration = step_data['duration']
            operations.append(Operation(job_name, step_num, machine, duration))
            step_num += 1

    operations.sort(key=lambda op: (int(op.job_name[3:]), op.step))

    job_completion_times = {}  # Track completion time of each job
    scheduled_operations = []

    for operation in operations:
        # Find the earliest possible start time for the operation
        start_time = 0

        if operation.job_name in job_completion_times:
            start_time = max(start_time, job_completion_times[operation.job_name])

        machine_availability[operation.machine] = operation.end_time

        # Update job completion time
        job_completion_times[operation.job_name] = operation.end_time

    # Calculate makespan
    makespan = max(job_completion_times.values())

    return makespan, scheduled_operations
